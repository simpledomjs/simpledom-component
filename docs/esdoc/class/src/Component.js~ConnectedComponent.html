    ConnectedComponent

    Extends:

    Component → ConnectedComponent

    Class for a component which react to store events.

      class Counter extends ConnectedComponent {
     eventsToSubscribe() {
         return [ 'UPDATE_COUNTER' ];
     }

     render() {
         return (
             <h1>{this.state.counter}</h1>
         );
     }
  }

  renderToDom('container', <Counter />);
